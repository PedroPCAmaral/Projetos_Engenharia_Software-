#include <stdio.h>

int main() {
    float notas[20], media = 0;
    int i, count = 0;

    for (i = 0; i < 20; i++) {
        printf("Digite a nota do aluno %d: ", i + 1);
        scanf("%f", &notas[i]);
        media += notas[i];
    }
    media /= 20;

    for (i = 0; i < 20; i++) {
        if (notas[i] > media) {
            count++;
        }
    }

    printf("Media da turma: %.2f\n", media);
    printf("Alunos com nota acima da media: %d\n", count);
    return 0;
}

int main() {
    float temperaturas[365], media = 0, menor, maior;
    int i, count = 0;

    for (i = 0; i < 365; i++) {
        printf("Digite a temperatura do dia %d: ", i + 1);
        scanf("%f", &temperaturas[i]);
        media += temperaturas[i];
    }
    media /= 365;

    menor = temperaturas[0];
    maior = temperaturas[0];

    for (i = 1; i < 365; i++) {
        if (temperaturas[i] < menor) {
            menor = temperaturas[i];
        }
        if (temperaturas[i] > maior) {
            maior = temperaturas[i];
        }
        if (temperaturas[i] < media) {
            count++;
        }
    }

    printf("Menor temperatura: %.2f\n", menor);
    printf("Maior temperatura: %.2f\n", maior);
    printf("Temperatura média anual: %.2f\n", media);
    printf("Dias com temperatura inferior à média: %d\n", count);
    return 0;
}

#include <stdio.h>

int main() {
    int VET[50], i, j, repetidos = 0;

    printf("Digite 50 numeros:\n");
    for (i = 0; i < 50; i++) {
        printf("Numero %d: ", i + 1);
        scanf("%d", &VET[i]);
    }

    printf("Numeros repetidos e suas posicoes:\n");
    for (i = 0; i < 50; i++) {
        for (j = i + 1; j < 50; j++) {
            if (VET[i] == VET[j]) {
                printf("Numero %d encontrado nas posicoes %d e %d\n", VET[i], i + 1, j + 1);
                repetidos = 1;
                break; // sair do loop interno se já encontrou um repetido
            }
        }
    }

    if (!repetidos) {
        printf("Nao ha numeros repetidos no vetor.\n");
    }
    return 0;
}

#include <stdio.h>

int main() {
    int matriz[3][3], i, j;
    int determinante;

    printf("Digite os elementos da matriz 3x3:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            printf("Elemento [%d][%d]: ", i + 1, j + 1);
            scanf("%d", &matriz[i][j]);
        }
    }

    // Cálculo do determinante
    determinante = matriz[0][0] * (matriz[1][1] * matriz[2][2] - matriz[1][2] * matriz[2][1]) -
                   matriz[0][1] * (matriz[1][0] * matriz[2][2] - matriz[1][2] * matriz[2][0]) +
                   matriz[0][2] * (matriz[1][0] * matriz[2][1] - matriz[1][1] * matriz[2][0]);

    printf("Determinante da matriz: %d\n", determinante);
    return 0;
}
